\section{Démarrage}

\subsection{Commande}

\begin{enumerate}
\item Ouvrir une fenêtre de commande.
\item Passer dans le répertoire \code{tetrahead}.
\item Exécuter la commande \code{java -jar tetrahead.jar}
\end{enumerate}

\subsection{Problèmes pouvant être rencontrés}
\index{Problemes@Problèmes}
\begin{description}
\item[Ne trouve pas l'exécutable :] passer le chemin complet de l'exécutable 
Java (cf. section~\ref{installation}, page~\pageref{installation}).

\item[Fichier tetrahead.jar corrompu :] télécharger à nouveau le logiciel \tth, 
puis le réinstaller (cf. section~\ref{installation}, 
page~\pageref{installation}).

\end{description}
